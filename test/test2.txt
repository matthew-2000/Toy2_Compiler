proc main():
    var a, b ^= 1, 3;\
    if yes() then
        a ^= a + b;
        -->! $(a);
    elseif yesdue() then
        a ^= a - b;
        -->! $(a);
    else
        -->! "No";
    endif;

    while yes() do
            -->! "No";
    endwhile;
endproc

func yes()-> boolean:
    return false;
endfunc

func yesdue()-> boolean:
    return true;
endfunc
